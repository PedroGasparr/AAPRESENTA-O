Técnicas de Otimização de Custos

Reduza seus gastos em até 60% com estas estratégias comprovadas

1 Cache Local com Persistência

Reduza leituras repetidas ativando a persistência offline:

// Ativar cache
firebase.firestore().enablePersistence()
  .then(() => console.log("Cache ativado"))
  .catch(err => {
    if (err.code == 'failed-precondition') {
      // Múltiplas abas abertas
    } else if (err.code == 'unimplemented') {
      // Navegador não suportado
    }
  });
Economia: Até 60% em leituras para apps com dados estáticos
2 Operações em Lote (Batch)

Agrupe múltiplas escritas em uma única operação:

// Exemplo: Atualizar 3 documentos com 1 operação
const batch = db.batch();
const userRef = db.collection('users').doc('user1');
const orderRef = db.collection('orders').doc('order1');
const logRef = db.collection('logs').doc();

batch.update(userRef, {lastActive: new Date()});
batch.delete(orderRef);
batch.set(logRef, {event: 'update', time: new Date()});

await batch.commit(); // 1 operação de escrita!
Economia: Até 80% em operações de escrita em atualizações em massa
3 Estrutura de Dados Eficiente

Evite aninhamento profundo e otimize seus documentos:

Não Recomendado:

{
  user: {
    posts: {
      post1: { ... },
      post2: { ... }
      // 100+ posts
    }
  }
}

Recomendado:

// Coleção separada
users/{userId}
posts/{postId}
  - userId: "123"
  - ...dados do post
Economia: Redução de 40-70% em leituras/escritas
4 Limpeza Automática de Dados

Implemente TTL (Time-To-Live) para dados temporários:

// Adicionar campo de expiração
await db.collection('logs').add({
  message: "Erro no sistema",
  createdAt: new Date(),
  expiresAt: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) // 30 dias
});

// Função Cloud para limpeza
exports.cleanOldLogs = functions.pubsub
  .schedule('every 24 hours')
  .onRun(async (context) => {
    const now = new Date();
    const oldLogs = await db.collection('logs')
      .where('expiresAt', '<=', now)
      .get();
    
    const batch = db.batch();
    oldLogs.docs.forEach(doc => batch.delete(doc.ref));
    await batch.commit();
  });
Economia: Redução de 20-50% em armazenamento

Como Implementar com Eficiência

Exemplo completo de implementação otimizada

// Configuração inicial otimizada
import { initializeApp } from 'firebase/app';
import { getFirestore, enableIndexedDbPersistence } from 'firebase/firestore';
import { getDatabase } from 'firebase/database';
import { getStorage } from 'firebase/storage';
import { getAuth } from 'firebase/auth';

const firebaseConfig = {
  apiKey: "SUA_CHAVE_API",
  authDomain: "SEU_DOMINIO.firebaseapp.com",
  databaseURL: "https://SEU_PROJETO.firebaseio.com",
  projectId: "SEU_PROJETO",
  storageBucket: "SEU_PROJETO.appspot.com",
  messagingSenderId: "SEU_SENDER_ID",
  appId: "SEU_APP_ID"
};

// Inicialize o Firebase
const app = initializeApp(firebaseConfig);

// Configure os serviços
const db = getFirestore(app);
const rtdb = getDatabase(app); // Realtime Database
const storage = getStorage(app);
const auth = getAuth(app);

// Ative a persistência para cache offline (Firestore)
enableIndexedDbPersistence(db)
  .catch((err) => {
    if (err.code == 'failed-precondition') {
      console.warn("Persistência falhou - Múltiplas abas abertas");
    } else if (err.code == 'unimplemented') {
      console.warn("Persistência não suportada pelo navegador");
    }
  });
// Autenticação com monitoramento de custos
import { 
  signInWithEmailAndPassword,
  createUserWithEmailAndPassword,
  onAuthStateChanged
} from 'firebase/auth';

// Cadastro otimizado
async function registerUser(email, password) {
  try {
    const userCredential = await createUserWithEmailAndPassword(
      auth, email, password
    );
    
    // Crie o documento do usuário em uma única operação
    await setDoc(doc(db, "users", userCredential.user.uid), {
      email: email,
      createdAt: serverTimestamp(),
      lastLogin: serverTimestamp(),
      plan: "free"
    });
    
    console.log("Usuário registrado com sucesso!");
  } catch (error) {
    console.error("Erro no registro:", error.message);
  }
}

// Monitorar estado de autenticação (single listener)
onAuthStateChanged(auth, (user) => {
  if (user) {
    console.log("Usuário logado:", user.uid);
    // Atualize lastLogin apenas uma vez por sessão
    updateDoc(doc(db, "users", user.uid), {
      lastLogin: serverTimestamp()
    });
  } else {
    console.log("Usuário deslogado");
  }
});
// Operações otimizadas no Firestore
import {
  collection, query, where,
  getDocs, getDoc, setDoc,
  updateDoc, deleteDoc,
  serverTimestamp, writeBatch
} from 'firebase/firestore';

// Busca paginada com limite
async function getPaginatedPosts(limit = 10, lastDoc = null) {
  let q = query(
    collection(db, "posts"),
    where("published", "==", true),
    orderBy("createdAt", "desc"),
    limit(limit)
  );
  
  if (lastDoc) {
    q = query(q, startAfter(lastDoc));
  }
  
  const snapshot = await getDocs(q);
  const posts = [];
  snapshot.forEach(doc => posts.push({ id: doc.id, ...doc.data() }));
  
  return {
    posts,
    lastDoc: snapshot.docs[snapshot.docs.length - 1]
  };
}

// Atualização em lote
async function updateUserStats(userId, stats) {
  const batch = writeBatch(db);
  const userRef = doc(db, "users", userId);
  const statsRef = doc(db, "stats", userId);
  const logRef = doc(collection(db, "logs"));
  
  batch.update(userRef, { lastActive: serverTimestamp() });
  batch.set(statsRef, stats, { merge: true });
  batch.set(logRef, {
    event: "stats_update",
    userId,
    timestamp: serverTimestamp()
  });
  
  await batch.commit();
}
// Operações otimizadas no Realtime Database
import { ref, set, update, onValue, off } from 'firebase/database';

// Escrever dados
async function writeUserData(userId, name, email) {
  try {
    await set(ref(rtdb, 'users/' + userId), {
      username: name,
      email: email,
      last_login: Date.now()
    });
    console.log("Dados escritos com sucesso!");
  } catch (error) {
    console.error("Erro ao escrever dados:", error);
  }
}

// Atualizar dados
async function updateUserProfile(userId, updates) {
  try {
    const updates = {};
    updates['/users/' + userId + '/last_updated'] = Date.now();
    updates['/users/' + userId + '/profile'] = {
      ...updates
    };
    
    await update(ref(rtdb), updates);
    console.log("Perfil atualizado com sucesso!");
  } catch (error) {
    console.error("Erro ao atualizar perfil:", error);
  }
}

// Ler dados com listener otimizado
function setupUserListener(userId, callback) {
  const userRef = ref(rtdb, 'users/' + userId);
  
  const listener = onValue(userRef, (snapshot) => {
    const data = snapshot.val();
    callback(data);
  });
  
  // Retorna função para remover listener
  return () => off(userRef, 'value', listener);
}

// Uso:
const removeListener = setupUserListener('123', (userData) => {
  console.log("Dados atualizados:", userData);
});
